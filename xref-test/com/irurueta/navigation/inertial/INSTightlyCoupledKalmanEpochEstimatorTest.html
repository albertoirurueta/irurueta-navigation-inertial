325     private static INSTightlyCoupledKalmanState estimate(
326             final List<GNSSMeasurement> measurements, final INSTightlyCoupledKalmanState previousState,
327             final double fx, final double fy, final double fz, final double previousLatitude,
328             final INSTightlyCoupledKalmanConfig config) throws AlgebraException {
329 
330         // Skew symmetric matrix of Earth rate
331         final var omegaIe = Utils.skewMatrix(new double[]{0.0, 0.0, Constants.EARTH_ROTATION_RATE});
332 
333         // SYSTEM PROPAGATION PHASE
334 
335         // 1. Determine transition matrix using (14.50) (first-order approx)
336         final var phiMatrix = Matrix.identity(17, 17);
337         phiMatrix.setSubmatrix(0, 0, 2, 2,
338                 phiMatrix.getSubmatrix(0, 0, 2, 2)
339                         .subtractAndReturnNew(omegaIe.multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS)));
340 
341         final var estCbeOld = previousState.getBodyToEcefCoordinateTransformationMatrix();
342         phiMatrix.setSubmatrix(0, 12, 2, 14,
343                 estCbeOld.multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS));
344 
345         final var measFibb = Matrix.newFromArray(new double[]{fx, fy, fz});
346         phiMatrix.setSubmatrix(3, 0, 5, 2,
347                 Utils.skewMatrix(estCbeOld.multiplyAndReturnNew(measFibb))
348                         .multiplyByScalarAndReturnNew(-TIME_INTERVAL_SECONDS));
349 
350         phiMatrix.setSubmatrix(3, 3, 5, 5,
351                 phiMatrix.getSubmatrix(3, 3, 5, 5)
352                         .subtractAndReturnNew(omegaIe.multiplyByScalarAndReturnNew(2.0 * TIME_INTERVAL_SECONDS)));
353 
354         final var sinPrevLat = Math.sin(previousLatitude);
355         final var cosPrevLat = Math.cos(previousLatitude);
356         final var sinPrevLat2 = sinPrevLat * sinPrevLat;
357         final var cosPrevLat2 = cosPrevLat * cosPrevLat;
358 
359         final var geocentricRadius = Constants.EARTH_EQUATORIAL_RADIUS_WGS84
360                 / Math.sqrt(1.0 - Math.pow(Constants.EARTH_ECCENTRICITY * sinPrevLat, 2.0))
361                 * Math.sqrt(cosPrevLat2
362                 + Math.pow(1.0 - Constants.EARTH_ECCENTRICITY * Constants.EARTH_ECCENTRICITY, 2.0)
363                 * sinPrevLat2);
364 
365         final var prevX = previousState.getX();
366         final var prevY = previousState.getY();
367         final var prevZ = previousState.getZ();
368         final var gravity = ECEFGravityEstimator.estimateGravityAndReturnNew(prevX, prevY, prevZ);
369         final var g = gravity.asMatrix();
370 
371         final var estRebeOld = Matrix.newFromArray(new double[]{prevX, prevY, prevZ});
372 
373         final var gScaled = g.multiplyByScalarAndReturnNew(-2.0 * TIME_INTERVAL_SECONDS / geocentricRadius);
374         final var estRebeOldTrans = estRebeOld.transposeAndReturnNew();
375         final var previousPositionNorm = Math.sqrt(prevX * prevX + prevY * prevY + prevZ * prevZ);
376         final var estRebeOldTransScaled = estRebeOldTrans.multiplyByScalarAndReturnNew(1.0 / previousPositionNorm);
377         phiMatrix.setSubmatrix(3, 6, 5, 8,
378                 gScaled.multiplyAndReturnNew(estRebeOldTransScaled));
379 
380         phiMatrix.setSubmatrix(3, 9, 5, 11,
381                 estCbeOld.multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS));
382 
383         phiMatrix.setSubmatrix(6, 3, 8, 5,
384                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS));
385 
386         phiMatrix.setElementAt(15, 16, TIME_INTERVAL_SECONDS);
387 
388         // 2. Determine approximate system noise covariance matrix using (14.82)
389         final var qPrimeMatrix = new Matrix(17, 17);
390         qPrimeMatrix.setSubmatrix(0, 0, 2, 2,
391                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
392                         config.getGyroNoisePSD() * TIME_INTERVAL_SECONDS));
393         qPrimeMatrix.setSubmatrix(3, 3, 5, 5,
394                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
395                         config.getAccelerometerNoisePSD() * TIME_INTERVAL_SECONDS));
396         qPrimeMatrix.setSubmatrix(9, 9, 11, 11,
397                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
398                         config.getAccelerometerBiasPSD() * TIME_INTERVAL_SECONDS));
399         qPrimeMatrix.setSubmatrix(12, 12, 14, 14,
400                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
401                         config.getGyroBiasPSD() * TIME_INTERVAL_SECONDS));
402         qPrimeMatrix.setElementAt(15, 15, config.getClockPhasePSD() * TIME_INTERVAL_SECONDS);
403         qPrimeMatrix.setElementAt(16, 16, config.getClockFrequencyPSD() * TIME_INTERVAL_SECONDS);
404 
405         // 3. Propagate state estimates using (3.14) noting that only the clock
406         // states are non-zero due to closed-loop correction.
407         final var xEstPropagated = new Matrix(17, 1);
408         xEstPropagated.setElementAtIndex(15, previousState.getReceiverClockOffset()
409                 + previousState.getReceiverClockDrift() * TIME_INTERVAL_SECONDS);
410         xEstPropagated.setElementAtIndex(16, previousState.getReceiverClockDrift());
411 
412         // 4. Propagate state estimation error covariance matrix using (3.46)
413         final var pMatrixOld = previousState.getCovariance();
414         final var phiTrans = phiMatrix.transposeAndReturnNew();
415         final var halfQ = qPrimeMatrix.multiplyByScalarAndReturnNew(0.5);
416         final var tmp1 = pMatrixOld.addAndReturnNew(halfQ);
417         final var pMatrixPropagated = phiMatrix.multiplyAndReturnNew(tmp1);
418         pMatrixPropagated.multiply(phiTrans);
419         pMatrixPropagated.add(halfQ);
420 
421         // MEASUREMENT UPDATE PHASE
422 
423         final var noMeas = measurements.size();
424         final var uAseT = new Matrix(noMeas, 3);
425         final var predMeas = new Matrix(noMeas, 2);
426 
427         final var prevVx = previousState.getVx();
428         final var prevVy = previousState.getVy();
429         final var prevVz = previousState.getVz();
430         final var estVebeOld = Matrix.newFromArray(new double[]{prevVx, prevVy, prevVz});
431 
432         // Loop measurements
433         for (var j = 0; j < noMeas; j++) {
434             // Predict approx range
435             final var measurement = measurements.get(j);
436             var deltaR = Matrix.newFromArray(new double[]{
437                     measurement.getX() - prevX,
438                     measurement.getY() - prevY,
439                     measurement.getZ() - prevZ
440             });
441 
442             final var approxRange = Math.sqrt(Utils.dotProduct(deltaR.transposeAndReturnNew(), deltaR));
443 
444             // Calculate frame rotation during signal transit time using (8.36)
445             final var cei = new Matrix(3, 3);
446             cei.fromArray(new double[]{
447                     1.0, Constants.EARTH_ROTATION_RATE * approxRange / GNSSKalmanEpochEstimator.SPEED_OF_LIGHT, 0.0,
448                     -Constants.EARTH_ROTATION_RATE * approxRange / GNSSKalmanEpochEstimator.SPEED_OF_LIGHT, 1.0, 0.0,
449                     0.0, 0.0, 1.0}, false);
450 
451             // Predict pseudo-range using (9.165)
452             final var measurementPosition = Matrix.newFromArray(new double[]{
453                     measurement.getX(), measurement.getY(), measurement.getZ()});
454             final var measurementVelocity = Matrix.newFromArray(new double[]{
455                     measurement.getVx(), measurement.getVy(), measurement.getVz()});
456             deltaR = cei.multiplyAndReturnNew(measurementPosition);
457             deltaR.subtract(estRebeOld);
458             final var range = Math.sqrt(Utils.dotProduct(deltaR.transposeAndReturnNew(), deltaR));
459             predMeas.setElementAt(j, 0, range + xEstPropagated.getElementAtIndex(15));
460 
461             // Predict line of sight
462             uAseT.setSubmatrix(j, 0, j, 2,
463                     deltaR.transposeAndReturnNew().multiplyByScalarAndReturnNew(1.0 / range));
464 
465             // Predict pseudo-range rate using (9.165)
466             final var rangeRate = uAseT.getSubmatrix(j, 0, j, 2).multiplyAndReturnNew(
467                     cei.multiplyAndReturnNew(measurementVelocity.addAndReturnNew(
468                             omegaIe.multiplyAndReturnNew(measurementPosition)))
469                             .subtractAndReturnNew(estVebeOld.addAndReturnNew(
470                                     omegaIe.multiplyAndReturnNew(estRebeOld))));
471 
472             predMeas.setElementAt(j, 1, rangeRate.getElementAtIndex(0)
473                     + xEstPropagated.getElementAtIndex(16));
474         }
475 
476         // 5. Set up a measurement matrix using (14.126)
477         final var hMatrix = new Matrix(2 * noMeas, 17);
478         hMatrix.setSubmatrix(0, 6, noMeas - 1, 8,
479                 uAseT.getSubmatrix(0, 0, noMeas - 1, 2));
480         final var ones = new double[noMeas];
481         Arrays.fill(ones, 1.0);
482         hMatrix.setSubmatrix(0, 15, noMeas - 1, 15, ones);
483         hMatrix.setSubmatrix(noMeas, 3, 2 * noMeas - 1, 5,
484                 uAseT.getSubmatrix(0, 0, noMeas - 1, 2));
485         hMatrix.setSubmatrix(noMeas, 16, 2 * noMeas - 1, 16, ones);
486 
487         // 6. Set up a measurement noise covariance matrix assuming all measurements
488         // are independent and have equal variance for a given measurement type.
489         final var rMatrix = new Matrix(2 * noMeas, 2 * noMeas);
490         rMatrix.setSubmatrix(0, 0, noMeas - 1, noMeas - 1,
491                 Matrix.identity(noMeas, noMeas).multiplyByScalarAndReturnNew(Math.pow(config.getPseudoRangeSD(), 2.0)));
492         rMatrix.setSubmatrix(0, noMeas, noMeas - 1, 2 * noMeas - 1,
493                 new Matrix(noMeas, noMeas));
494         rMatrix.setSubmatrix(noMeas, 0, 2 * noMeas - 1, noMeas - 1,
495                 new Matrix(noMeas, noMeas));
496         rMatrix.setSubmatrix(noMeas, noMeas, 2 * noMeas - 1, 2 * noMeas - 1,
497                 Matrix.identity(noMeas, noMeas).multiplyByScalarAndReturnNew(Math.pow(config.getRangeRateSD(), 2.0)));
498 
499         // 7. Calculate Kalman gain using (3.21)
500         final var kMatrix = pMatrixPropagated.multiplyAndReturnNew(
501                 hMatrix.transposeAndReturnNew().multiplyAndReturnNew(
502                         Utils.inverse(hMatrix.multiplyAndReturnNew(pMatrixPropagated)
503                                 .multiplyAndReturnNew(hMatrix.transposeAndReturnNew())
504                                 .addAndReturnNew(rMatrix))));
505 
506         // 8. Formulate measurement innovations using (14.119)
507         final var deltaZ = new Matrix(2 * noMeas, 1);
508         for (var j = 0; j < noMeas; j++) {
509             final var measurement = measurements.get(j);
510             deltaZ.setElementAtIndex(j, measurement.getPseudoRange() - predMeas.getElementAtIndex(j));
511             deltaZ.setElementAtIndex(noMeas + j,
512                     measurement.getPseudoRate() - predMeas.getElementAtIndex(noMeas + j));
513         }
514 
515         // 9. Update state estimates using (3.24)
516         final var xEstNew = xEstPropagated.addAndReturnNew(kMatrix.multiplyAndReturnNew(deltaZ));
517 
518         // 10. Update state estimation error covariance matrix using (3.25)
519         final var pMatrixNew = Matrix.identity(17, 17)
520                 .subtractAndReturnNew(kMatrix.multiplyAndReturnNew(hMatrix))
521                 .multiplyAndReturnNew(pMatrixPropagated);
522 
523         // CLOSED-LOOP CORRECTION
524 
525         // Correct attitude, velocity, and position using (14.7-9)
526         final var estCbeNew = Matrix.identity(3, 3).subtractAndReturnNew(Utils.skewMatrix(
527                 xEstNew.getSubmatrix(0, 0, 2, 0)));
528         estCbeNew.multiply(estCbeOld);
529         final var estVebeNew = estVebeOld.subtractAndReturnNew(
530                 xEstNew.getSubmatrix(3, 0, 5, 0));
531         final var estRebeNew = estRebeOld.subtractAndReturnNew(
532                 xEstNew.getSubmatrix(6, 0, 8, 0));
533 
534         // Update IMU bias and GNSS receiver clock estimates
535         final var estIMUbiasOld = Matrix.newFromArray(new double[]{
536                 previousState.getAccelerationBiasX(),
537                 previousState.getAccelerationBiasY(),
538                 previousState.getAccelerationBiasZ(),
539                 previousState.getGyroBiasX(),
540                 previousState.getGyroBiasY(),
541                 previousState.getGyroBiasZ()
542         });
543         final var estIMUbiasNew = estIMUbiasOld.addAndReturnNew(
544                 xEstNew.getSubmatrix(9, 0, 14, 0));
545 
546         final var estClockNew = xEstNew.getSubmatrix(
547                 15, 0, 16, 0);
548 
549         return new INSTightlyCoupledKalmanState(estCbeNew,
550                 estVebeNew.getElementAtIndex(0),
551                 estVebeNew.getElementAtIndex(1),
552                 estVebeNew.getElementAtIndex(2),
553                 estRebeNew.getElementAtIndex(0),
554                 estRebeNew.getElementAtIndex(1),
555                 estRebeNew.getElementAtIndex(2),
556                 estIMUbiasNew.getElementAtIndex(0),
557                 estIMUbiasNew.getElementAtIndex(1),
558                 estIMUbiasNew.getElementAtIndex(2),
559                 estIMUbiasNew.getElementAtIndex(3),
560                 estIMUbiasNew.getElementAtIndex(4),
561                 estIMUbiasNew.getElementAtIndex(5),
562                 estClockNew.getElementAtIndex(0),
563                 estClockNew.getElementAtIndex(1),
564                 pMatrixNew);
565     }
