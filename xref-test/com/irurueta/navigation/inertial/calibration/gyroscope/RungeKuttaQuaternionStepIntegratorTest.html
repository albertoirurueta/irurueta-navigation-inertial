38      private static final double TIME_INTERVAL = 0.02;

40      private static final int NUM_SAMPLES = 50;

44      private static final double MIN_ANGULAR_SPEED_DEGREES_PER_SECOND = -45.0;

46      private static final double MAX_ANGULAR_SPEED_DEGREES_PER_SECOND = 45.0;

53  
54      private static final double SMALL_ABSOLUTE_ERROR = 2e-4;

56      @Test
57      void integrate_whenOneStep_computesExpectedResult() throws RotationException {
58          final var randomizer = new UniformRandomizer();
59          final var wx = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
60                  MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
61          final var wy = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
62                  MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
63          final var wz = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
64                  MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
65  
66          final var initialAttitude = getInitialAttitude();
67          final var totalTime = TIME_INTERVAL;
68          final var delta = new Quaternion(totalTime * wx, totalTime * wy, totalTime * wz);
69          final var expectedResult = initialAttitude.combineAndReturnNew(delta);
70          expectedResult.normalize();
71  
72          final var integrator = new RungeKuttaQuaternionStepIntegrator();
73          final var result1 = new Quaternion();
74          final var result2 = new Quaternion();
75          RungeKuttaQuaternionStepIntegrator.integrationStep(initialAttitude, wx, wy, wz, wx, wy, wz, TIME_INTERVAL,
76                  result1);
77          integrator.integrate(initialAttitude, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result2);
78  
79          assertEquals(result1, result2);
80          assertTrue(expectedResult.equals(result1, SMALL_ABSOLUTE_ERROR));
81      }

83      @Test
84      void integrate_whenMultipleSteps_computesExpectedResult() throws RotationException {
85          var numValid = 0;
86          for (var t = 0; t < TIMES; t++) {
87              final var randomizer = new UniformRandomizer();
88              final var wx = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
89                      MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
90              final var wy = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
91                      MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
92              final var wz = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
93                      MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
94  
95              final var initialAttitude = getInitialAttitude();
96              final var totalTime = NUM_SAMPLES * TIME_INTERVAL;
97              final var delta = new Quaternion(totalTime * wx, totalTime * wy, totalTime * wz);
98              final var expectedResult = initialAttitude.combineAndReturnNew(delta);
99              expectedResult.normalize();
100 
101             final var integrator = new RungeKuttaQuaternionStepIntegrator();
102             final var result1 = new Quaternion(initialAttitude);
103             final var result2 = new Quaternion(initialAttitude);
104             for (var i = 0; i < NUM_SAMPLES; i++) {
105                 RungeKuttaQuaternionStepIntegrator.integrationStep(result1, wx, wy, wz, wx, wy, wz, TIME_INTERVAL,
106                         result1);
107                 integrator.integrate(result2, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result2);
108             }
109 
110             assertEquals(result1, result2);
111 
112             final var resultEulerAngles = result1.toEulerAngles();
113             final var expectedEulerAngles = expectedResult.toEulerAngles();
114             final var diffAngles = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultEulerAngles);
115             Logger.getLogger(RungeKuttaQuaternionStepIntegratorTest.class.getName()).log(Level.INFO, 
116                     String.format("Error euler angles: %s",  Arrays.toString(diffAngles)));
117 
118             if (!expectedResult.equals(result1, ABSOLUTE_ERROR)) {
119                 continue;
120             }
121             assertTrue(expectedResult.equals(result1, ABSOLUTE_ERROR));
122             numValid++;
123             break;
124         }
125 
126         assertTrue(numValid > 0);
127     }

129     @Test
130     void compareErrors_whenNoNoise_haveExpectedRelativeAccuracy() throws RotationException {
131         var numValid = 0;
132         var avgError1 = 0.0;
133         var avgError2 = 0.0;
134         var avgError3 = 0.0;
135         var avgError4 = 0.0;
136         var avgError5 = 0.0;
137         var avgError6 = 0.0;
138         var num = 0;
139         for (var t = 0; t < TIMES; t++){
140             final var randomizer = new UniformRandomizer();
141             final var wx = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
142                     MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
143             final var wy = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
144                     MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
145             final var wz = Math.toRadians(randomizer.nextDouble(MIN_ANGULAR_SPEED_DEGREES_PER_SECOND,
146                     MAX_ANGULAR_SPEED_DEGREES_PER_SECOND));
147 
148             final var initialAttitude = getInitialAttitude();
149             final var totalTime = NUM_SAMPLES * TIME_INTERVAL;
150             final var delta = new Quaternion(totalTime * wx, totalTime * wy, totalTime * wz);
151             final var expectedResult = initialAttitude.combineAndReturnNew(delta);
152             expectedResult.normalize();
153 
154             final var suhIntegrator = new SuhQuaternionStepIntegrator();
155             final var trawnyIntegrator = new TrawnyQuaternionStepIntegrator();
156             final var yuanIntegrator = new YuanQuaternionStepIntegrator();
157             final var eulerIntegrator = new EulerQuaternionStepIntegrator();
158             final var midPointIntegrator = new MidPointQuaternionStepIntegrator();
159             final var rungeKuttaIntegrator = new RungeKuttaQuaternionStepIntegrator();
160             final var result1 = new Quaternion(initialAttitude);
161             final var result2 = new Quaternion(initialAttitude);
162             final var result3 = new Quaternion(initialAttitude);
163             final var result4 = new Quaternion(initialAttitude);
164             final var result5 = new Quaternion(initialAttitude);
165             final var result6 = new Quaternion(initialAttitude);
166             for (var i = 0; i < NUM_SAMPLES; i++) {
167                 suhIntegrator.integrate(result1, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result1);
168                 trawnyIntegrator.integrate(result2, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result2);
169                 yuanIntegrator.integrate(result3, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result3);
170                 eulerIntegrator.integrate(result1, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result4);
171                 midPointIntegrator.integrate(result2, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result5);
172                 rungeKuttaIntegrator.integrate(result3, wx, wy, wz, wx, wy, wz, TIME_INTERVAL, result6);
173             }
174 
175             final var resultSuhAngles = result1.toEulerAngles();
176             final var resultTrawnyEulerAngles = result2.toEulerAngles();
177             final var resultYuanAngles = result2.toEulerAngles();
178             final var resultEulerAngles = result1.toEulerAngles();
179             final var resultMidPointEulerAngles = result2.toEulerAngles();
180             final var resultRungeKuttaEulerAngles = result3.toEulerAngles();
181 
182             final var expectedEulerAngles = expectedResult.toEulerAngles();
183             final var diffAngles1 = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultSuhAngles);
184             final var diffAngles2 = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultTrawnyEulerAngles);
185             final var diffAngles3 = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultYuanAngles);
186             final var diffAngles4 = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultEulerAngles);
187             final var diffAngles5 = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultMidPointEulerAngles);
188             final var diffAngles6 = ArrayUtils.subtractAndReturnNew(expectedEulerAngles, resultRungeKuttaEulerAngles);
189 
190             final var error1 = Utils.normF(diffAngles1);
191             final var error2 = Utils.normF(diffAngles2);
192             final var error3 = Utils.normF(diffAngles3);
193             final var error4 = Utils.normF(diffAngles4);
194             final var error5 = Utils.normF(diffAngles5);
195             final var error6 = Utils.normF(diffAngles6);
196 
197             Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
198                     String.format("Suh error: %f radians", error1));
199             Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
200                     String.format("Trawny error: %f radians", error2));
201             Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
202                     String.format("Yuan error: %f radians", error3));
203             Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
204                     String.format("Euler error: %f radians", error4));
205             Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
206                     String.format("Mid-point error: %f radians", error5));
207             Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
208                     String.format("Runge-Kutta error: %f radians", error6));
209 
210             avgError1 += error1;
211             avgError2 += error2;
212             avgError3 += error3;
213             avgError4 += error4;
214             avgError5 += error5;
215             avgError6 += error6;
216             num++;
217 
218             if (error1 < error2 || error1 < error3 || error1 < error4 || error1 < error5 || error1 < error6) {
219                 continue;
220             }
221 
222             assertTrue(error1 >= error2);
223             assertTrue(error1 >= error3);
224             assertTrue(error1 >= error4);
225             assertTrue(error1 >= error5);
226             assertTrue(error1 >= error6);
227             numValid++;
228             break;
229         }
230 
231         assertTrue(numValid > 0);
232 
233         // compute ranking of errors
234         avgError1 /= num;
235         avgError2 /= num;
236         avgError3 /= num;
237         avgError4 /= num;
238         avgError5 /= num;
239         avgError6 /= num;
240 
241         final var map = new HashMap<String, Double>();
242         map.put("Suh", avgError1);
243         map.put("Trawny", avgError2);
244         map.put("Yuan", avgError3);
245         map.put("Euler", avgError4);
246         map.put("Mid-point", avgError5);
247         map.put("Runga-Kutta", avgError6);
248 
249         final var methodsList = new ArrayList<>(map.entrySet()).stream()
250                 .sorted(Map.Entry.comparingByValue())
251                 .map(entry -> entry.getKey() + ": " + entry.getValue())
252                 .toList();
253         final var methods = String.join(", ", methodsList);
254         Logger.getLogger(SuhQuaternionStepIntegratorTest.class.getName()).log(Level.INFO,
255                 String.format("Methods (ordered more accurate to less accurate): %s", methods));
256     }

258     private static Quaternion getInitialAttitude() {
259         final var randomizer = new UniformRandomizer();
260         final var roll = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
261         final var pitch = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
262         final var yaw = Math.toRadians(randomizer.nextDouble(MIN_ANGLE_DEGREES, MAX_ANGLE_DEGREES));
263         return new Quaternion(roll, pitch, yaw);
264     }
