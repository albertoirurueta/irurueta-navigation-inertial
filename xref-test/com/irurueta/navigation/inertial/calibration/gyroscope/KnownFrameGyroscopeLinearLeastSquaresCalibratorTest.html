463     @Test
464     void testCalibrateMultipleOrientationsForGeneralCaseWithMinimumMeasuresAndNoNoise() throws WrongSizeException, 
465             InvalidSourceAndDestinationFrameTypeException, LockedException, NotReadyException, CalibrationException {
466 
467         final var ba = generateBa();
468         final var bg = generateBg();
469         final var ma = generateMaGeneral();
470         final var mg = generateMg();
471         final var gg = generateGg();
472         // when using minimum number of measurements we must not add any noise so that
473         // a solution is found. When adding more measurements, certain noise can be added
474         final var accelNoiseRootPSD = 0.0;
475         final var gyroNoiseRootPSD = 0.0;
476         final var accelQuantLevel = 0.0;
477         final var gyroQuantLevel = 0.0;
478 
479         final var errors = new IMUErrors(ba, bg, ma, mg, gg, accelNoiseRootPSD, gyroNoiseRootPSD,
480                 accelQuantLevel, gyroQuantLevel);
481         
482         final var randomizer = new UniformRandomizer();
483         final var latitude = Math.toRadians(randomizer.nextDouble(MIN_LATITUDE_DEGREES, MAX_LATITUDE_DEGREES));
484         final var longitude = Math.toRadians(randomizer.nextDouble(MIN_LONGITUDE_DEGREES, MAX_LONGITUDE_DEGREES));
485         final var height = randomizer.nextDouble(MIN_HEIGHT, MAX_HEIGHT);
486         final var nedPosition = new NEDPosition(latitude, longitude, height);
