34      @Test
35      void testConstructor() {
36          // test empty constructor
37          var stdMagnetic = new StandardDeviationBodyMagneticFluxDensity();
38  
39          // check default values
40          assertNull(stdMagnetic.getMagneticFluxDensity());
41          assertEquals(0.0, stdMagnetic.getMagneticFluxDensityStandardDeviation(), 0.0);
42  
43          // test constructor with magnetic flux density
44          final var magneticFluxDensity = new BodyMagneticFluxDensity();
45          stdMagnetic = new StandardDeviationBodyMagneticFluxDensity(magneticFluxDensity);
46  
47          // check default values
48          assertSame(magneticFluxDensity, stdMagnetic.getMagneticFluxDensity());
49          assertEquals(0.0, stdMagnetic.getMagneticFluxDensityStandardDeviation(), 0.0);
50  
51          // test constructor with standard deviation
52          final var randomizer = new UniformRandomizer();
53          final var std = randomizer.nextDouble(0.0, MAX_MAGNETIC_FLUX_DENSITY);
54          stdMagnetic = new StandardDeviationBodyMagneticFluxDensity(std);
55  
56          // check default values
57          assertNull(stdMagnetic.getMagneticFluxDensity());
58          assertEquals(std, stdMagnetic.getMagneticFluxDensityStandardDeviation(), 0.0);
59  
60          // Force IllegalArgumentException
61          assertThrows(IllegalArgumentException.class, () -> new StandardDeviationBodyMagneticFluxDensity(
62                  -1.0));
63  
64          // test constructor with magnetic flux density and standard deviation
65          stdMagnetic = new StandardDeviationBodyMagneticFluxDensity(magneticFluxDensity, std);
66  
67          // check default values
68          assertSame(magneticFluxDensity, stdMagnetic.getMagneticFluxDensity());
69          assertEquals(std, stdMagnetic.getMagneticFluxDensityStandardDeviation(), 0.0);
70  
71          // Force IllegalArgumentException
72          assertThrows(IllegalArgumentException.class, () -> new StandardDeviationBodyMagneticFluxDensity(
73                  magneticFluxDensity, -1.0));
74  
75          // test copy constructor
76          stdMagnetic = new StandardDeviationBodyMagneticFluxDensity(magneticFluxDensity, std);
77  
78          final var stdMagnetic2 = new StandardDeviationBodyMagneticFluxDensity(
79                  stdMagnetic);
80  
81          // check
82          assertEquals(stdMagnetic.getMagneticFluxDensity(), stdMagnetic2.getMagneticFluxDensity());
83          assertEquals(stdMagnetic.getMagneticFluxDensityStandardDeviation(), 
84                  stdMagnetic2.getMagneticFluxDensityStandardDeviation(), 0.0);
85      }
