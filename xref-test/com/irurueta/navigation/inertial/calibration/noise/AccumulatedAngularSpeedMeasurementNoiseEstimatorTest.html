77          assertEquals(0.0, estimator.getVariance(), 0.0);
78          assertEquals(0.0, estimator.getStandardDeviation(), 0.0);
79          final var std1 = estimator.getStandardDeviationAsMeasurement();
80          assertEquals(0.0, std1.getValue().doubleValue(), 0.0);

87          assertEquals(0.0, estimator.getGyroscopeBaseNoiseLevelRootPsd(), 0.0);

186     @Test
187     void testAddTriadAndReset1() throws LockedException, WrongSizeException {
188         final var ba = generateBa();
189         final var bg = generateBg();
190         final var ma = generateMa();
191         final var mg = generateMg();
192         final var gg = generateGg();
193         final var accelNoiseRootPSD = getAccelNoiseRootPsd();
194         final var gyroNoiseRootPSD = getGyroNoiseRootPsd();
195         final var accelQuantLevel = 0.0;
196         final var gyroQuantLevel = 0.0;
197 
198         final var errors = new IMUErrors(ba, bg, ma, mg, gg, accelNoiseRootPSD, gyroNoiseRootPSD, accelQuantLevel,
199                 gyroQuantLevel);
200         
201         final var randomizer = new UniformRandomizer();
202         final var fx = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE, MAX_ACCELEROMETER_VALUE);
203         final var fy = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE, MAX_ACCELEROMETER_VALUE);
204         final var fz = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE, MAX_ACCELEROMETER_VALUE);
205         final var omegaX = 0.0;
206         final var omegaY = 0.0;
207         final var omegaZ = 0.0;
208 
209         final var trueKinematics = new BodyKinematics(fx, fy, fz, omegaX, omegaY, omegaZ);
210 
211         final var estimator = new AccumulatedAngularSpeedMeasurementNoiseEstimator(this);
212 
213         reset();
214         assertEquals(0, start);
215         assertEquals(0, measurementAdded);
216         assertEquals(0, reset);
217         assertEquals(0, estimator.getNumberOfProcessedSamples());
218         assertNull(estimator.getLastMeasurement());
219         assertFalse(estimator.getLastMeasurement(null));
220         assertFalse(estimator.isRunning());
221 
222         final var triad = new AngularSpeedTriad();
223         final var kinematics = new BodyKinematics();
224         final var timeInterval = estimator.getTimeInterval();
225         final var lastMeasurement = new AngularSpeed(0.0, AngularSpeedUnit.DEGREES_PER_SECOND);
226         double value;
227         var avg = 0.0;
228         var v = 0.0;
229         final var random = new Random();
230         for (int i = 0, j = 1; i < N_SAMPLES; i++, j++) {
231             if (estimator.getLastMeasurement(lastMeasurement)) {
232                 assertEquals(lastMeasurement, estimator.getLastMeasurement());
233                 assertEquals(lastMeasurement, triad.getMeasurementNorm());
234             }
235 
236             BodyKinematicsGenerator.generate(timeInterval, trueKinematics, errors, random, kinematics);
237             kinematics.getAngularRateTriad(triad);
238 
239             value = triad.getNorm();
240 
241             estimator.addMeasurement(value);
242 
243             assertTrue(estimator.getLastMeasurement(lastMeasurement));
244             assertEquals(lastMeasurement, triad.getMeasurementNorm());
245             assertEquals(i + 1, estimator.getNumberOfProcessedSamples());
246             assertFalse(estimator.isRunning());
247 
248             avg = avg * (double) i / (double) j + value / j;
249 
250             final var diff = value - avg;
251 
252             final var diff2 = diff * diff;
253 
254             v = v * (double) i / (double) j + diff2 / j;
255         }
256 
257         assertEquals(N_SAMPLES, estimator.getNumberOfProcessedSamples());
258         assertFalse(estimator.isRunning());
259         assertEquals(1, start);
260         assertEquals(N_SAMPLES, measurementAdded);
261         assertEquals(0, reset);
262 
263         assertEquals(avg, estimator.getAvg(), ABSOLUTE_ERROR);
264 
265         final var avg1 = estimator.getAvgAsMeasurement();
266         assertEquals(avg, avg1.getValue().doubleValue(), ABSOLUTE_ERROR);
267         assertEquals(AngularSpeedUnit.RADIANS_PER_SECOND, avg1.getUnit());
268         final var avg2 = new AngularSpeed(1.0, AngularSpeedUnit.DEGREES_PER_SECOND);
269         estimator.getAvgAsMeasurement(avg2);
270         assertEquals(avg1, avg2);
271 
272         assertEquals(v, estimator.getVariance(), ABSOLUTE_ERROR);
273 
274         final var std = Math.sqrt(v);
275 
276         assertEquals(std, estimator.getStandardDeviation(), ABSOLUTE_ERROR);
277 
278         final var std1 = estimator.getStandardDeviationAsMeasurement();
279         assertEquals(std, std1.getValue().doubleValue(), ABSOLUTE_ERROR);
280         assertEquals(AngularSpeedUnit.RADIANS_PER_SECOND, std1.getUnit());
281         final var std2 = new AngularSpeed(1.0, AngularSpeedUnit.DEGREES_PER_SECOND);
282         estimator.getStandardDeviationAsMeasurement(std2);
283         assertEquals(std1, std2);
284 
285         final var psd = timeInterval * v;
286 
287         assertEquals(psd, estimator.getPsd(), ABSOLUTE_ERROR);
288 
289         final var rootPsd = Math.sqrt(psd);
290 
291         assertEquals(rootPsd, estimator.getRootPsd(), ABSOLUTE_ERROR);
292         assertEquals(estimator.getRootPsd(), estimator.getGyroscopeBaseNoiseLevelRootPsd(), 0.0);
293 
294         // reset
295         assertTrue(estimator.reset());
296 
297         assertEquals(1, reset);
298 
299         assertNull(estimator.getLastMeasurement());
300         assertFalse(estimator.getLastMeasurement(null));
301         assertEquals(0.0, estimator.getAvg(), 0.0);
302         assertEquals(0.0, estimator.getVariance(), 0.0);
303         assertEquals(0.0, estimator.getStandardDeviation(), 0.0);
304         assertEquals(0.0, estimator.getPsd(), 0.0);
305         assertEquals(0.0, estimator.getRootPsd(), 0.0);
306         assertEquals(0.0, estimator.getGyroscopeBaseNoiseLevelRootPsd(), 0.0);
307         assertEquals(0, estimator.getNumberOfProcessedSamples());
308         assertFalse(estimator.isRunning());
309     }

436     @Override
437     public void onStart(final AccumulatedAngularSpeedMeasurementNoiseEstimator estimator) {
438         checkLocked(estimator);
439         start++;
440     }
441 
442     @Override
443     public void onMeasurementAdded(final AccumulatedAngularSpeedMeasurementNoiseEstimator estimator) {
444         measurementAdded++;
445     }
446 
447     @Override
448     public void onReset(final AccumulatedAngularSpeedMeasurementNoiseEstimator estimator) {
449         reset++;
450     }
451 
452     private void reset() {
453         start = 0;
454         measurementAdded = 0;
455         reset = 0;
456     }
457 
458     private void checkLocked(final AccumulatedAngularSpeedMeasurementNoiseEstimator estimator) {
459         assertTrue(estimator.isRunning());
460         assertThrows(LockedException.class, () -> estimator.setTimeInterval(0.0));
461         assertThrows(LockedException.class, () -> estimator.setTimeInterval(new Time(0.0, TimeUnit.SECOND)));
462         assertThrows(LockedException.class, () -> estimator.setListener(this));
463         assertThrows(LockedException.class, () -> estimator.addMeasurement(0.0));
464         final var w = new AngularSpeed(0.0, AngularSpeedUnit.RADIANS_PER_SECOND);
465         assertThrows(LockedException.class, () -> estimator.addMeasurement(w));
466         assertThrows(LockedException.class, estimator::reset);
467     }

469     private static Matrix generateBa() {
470         return Matrix.newFromArray(new double[]{
471                 900 * MICRO_G_TO_METERS_PER_SECOND_SQUARED,
472                 -1300 * MICRO_G_TO_METERS_PER_SECOND_SQUARED,
473                 800 * MICRO_G_TO_METERS_PER_SECOND_SQUARED});
474     }
475 
476     private static Matrix generateBg() {
477         return Matrix.newFromArray(new double[]{
478                 -9 * DEG_TO_RAD / 3600.0,
479                 13 * DEG_TO_RAD / 3600.0,
480                 -8 * DEG_TO_RAD / 3600.0});
481     }
482 
483     private static Matrix generateMa() throws WrongSizeException {
484         final var result = new Matrix(3, 3);
485         result.fromArray(new double[]{
486                 500e-6, -300e-6, 200e-6,
487                 -150e-6, -600e-6, 250e-6,
488                 -250e-6, 100e-6, 450e-6
489         }, false);
490 
491         return result;
492     }
493 
494     private Matrix generateMg() throws WrongSizeException {
495         final var result = new Matrix(3, 3);
496         result.fromArray(new double[]{
497                 400e-6, -300e-6, 250e-6,
498                 0.0, -300e-6, -150e-6,
499                 0.0, 0.0, -350e-6
500         }, false);
501 
502         return result;
503     }
504 
505     private Matrix generateGg() throws WrongSizeException {
506         final var result = new Matrix(3, 3);
507         final var tmp = DEG_TO_RAD / (3600 * 9.80665);
508         result.fromArray(new double[]{
509                 0.9 * tmp, -1.1 * tmp, -0.6 * tmp,
510                 -0.5 * tmp, 1.9 * tmp, -1.6 * tmp,
511                 0.3 * tmp, 1.1 * tmp, -1.3 * tmp
512         }, false);
513 
514         return result;
515     }
516 
517     private static double getAccelNoiseRootPsd() {
518         return 100.0 * MICRO_G_TO_METERS_PER_SECOND_SQUARED;
519     }
520 
521     private static double getGyroNoiseRootPsd() {
522         return 0.01 * DEG_TO_RAD / 60.0;
523     }
