219     @Test
220     void testAddMeasurementAndProcessAndThenReset1() throws WrongSizeException, LockedException {
221         final var ba = generateBa();
222         final var bg = generateBg();
223         final var ma = generateMa();
224         final var mg = generateMg();
225         final var gg = generateGg();
226         final var accelNoiseRootPSD = getAccelNoiseRootPsd();
227         final var gyroNoiseRootPSD = getGyroNoiseRootPsd();
228         final var accelQuantLevel = 0.0;
229         final var gyroQuantLevel = 0.0;
230 
231         final var errors = new IMUErrors(ba, bg, ma, mg, gg, accelNoiseRootPSD, gyroNoiseRootPSD, accelQuantLevel,
232                 gyroQuantLevel);
233         
234         final var randomizer = new UniformRandomizer();
235         final var fx = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE, MAX_ACCELEROMETER_VALUE);
236         final var fy = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE, MAX_ACCELEROMETER_VALUE);
237         final var fz = randomizer.nextDouble(MIN_ACCELEROMETER_VALUE, MAX_ACCELEROMETER_VALUE);
238         final var omegaX = 0.0;
239         final var omegaY = 0.0;
240         final var omegaZ = 0.0;
241 
242         final var trueKinematics = new BodyKinematics(fx, fy, fz, omegaX, omegaY, omegaZ);
243 
244         final var estimator = new WindowedAccelerationMeasurementNoiseEstimator(this);
245 
246         reset();
247         assertEquals(0, start);
248         assertEquals(0, measurementAdded);
249         assertEquals(0, windowFilled);
250         assertEquals(0, reset);
251         assertFalse(estimator.isWindowFilled());
252         assertEquals(0, estimator.getNumberOfProcessedSamples());
253         assertEquals(0, estimator.getNumberOfAddedSamples());
254         assertEquals(0, estimator.getNumberOfSamplesInWindow());
255         assertNull(estimator.getFirstWindowedMeasurementValue());
256         assertNull(estimator.getFirstWindowedMeasurement());
257         assertFalse(estimator.getFirstWindowedMeasurement(null));
258         assertNull(estimator.getLastWindowedMeasurementValue());
259         assertNull(estimator.getLastWindowedMeasurement());
260         assertFalse(estimator.getLastWindowedMeasurement(null));
261         assertFalse(estimator.isRunning());
262 
263         final var kinematics = new BodyKinematics();
264         final var firstKinematics = new BodyKinematics();
265         final var windowSize = estimator.getWindowSize();
266         final var timeInterval = estimator.getTimeInterval();
267         final var firstMeasurement = new Acceleration(0.0, AccelerationUnit.METERS_PER_SQUARED_SECOND);
268         final var lastMeasurement = new Acceleration(0.0, AccelerationUnit.METERS_PER_SQUARED_SECOND);
269         final var measurement = new Acceleration(0.0, AccelerationUnit.METERS_PER_SQUARED_SECOND);
270         double value;
271         double avg = 0.0;
272         final var measurements = new ArrayList<Acceleration>();
273         final var random = new Random();
274         for (var i = 0; i < windowSize; i++) {
275             if (estimator.getFirstWindowedMeasurement(firstMeasurement)) {
276                 assertEquals(firstMeasurement, estimator.getFirstWindowedMeasurement());
277                 assertEquals(firstMeasurement.getValue().doubleValue(), estimator.getFirstWindowedMeasurementValue(),
278                         0.0);
279                 assertEquals(firstMeasurement, firstKinematics.getSpecificForceNormAsAcceleration());
280             }
281             if (estimator.getLastWindowedMeasurement(lastMeasurement)) {
282                 assertEquals(lastMeasurement, estimator.getLastWindowedMeasurement());
283                 assertEquals(lastMeasurement.getValue().doubleValue(), estimator.getLastWindowedMeasurementValue(),
284                         0.0);
285                 assertEquals(lastMeasurement, measurement);
286             }
287 
288             BodyKinematicsGenerator.generate(timeInterval, trueKinematics, errors, random, kinematics);
289 
290             if (i == 0) {
291                 firstKinematics.copyFrom(kinematics);
292             }
293 
294             kinematics.getSpecificForceNormAsAcceleration(measurement);
295             measurements.add(new Acceleration(measurement.getValue(), measurement.getUnit()));
296             value = measurement.getValue().doubleValue();
297 
298             estimator.addMeasurementAndProcess(value);
299 
300             assertTrue(estimator.getLastWindowedMeasurement(lastMeasurement));
301             assertEquals(lastMeasurement, measurement);
302             assertEquals(i + 1, estimator.getNumberOfProcessedSamples());
303             assertEquals(i + 1, estimator.getNumberOfAddedSamples());
304             assertEquals(i + 1, estimator.getNumberOfSamplesInWindow());
305             assertFalse(estimator.isRunning());
306 
307             avg += value;
308         }
309 
310         avg /= windowSize;
311 
312         var v = 0.0;
313         for (var i = 0; i < windowSize; i++) {
314             measurement.setValue(measurements.get(i).getValue());
315             measurement.setUnit(measurements.get(i).getUnit());
316 
317             value = measurement.getValue().doubleValue();
318 
319             final var diff = value - avg;
320 
321             v += diff * diff;
322         }
323 
324         v /= (windowSize - 1);
325 
326         final var std = Math.sqrt(v);
327 
328         assertEquals(avg, estimator.getAvg(), ABSOLUTE_ERROR);
329 
330         final var a1 = estimator.getAvgAsMeasurement();
331         assertEquals(avg, a1.getValue().doubleValue(), ABSOLUTE_ERROR);
332         assertEquals(AccelerationUnit.METERS_PER_SQUARED_SECOND, a1.getUnit());
333         final var a2 = new Acceleration(0.0, AccelerationUnit.FEET_PER_SQUARED_SECOND);
334         estimator.getAvgAsMeasurement(a2);
335         assertEquals(a1, a2);
336 
337         assertEquals(v, estimator.getVariance(), ABSOLUTE_ERROR);
338         assertEquals(std, estimator.getStandardDeviation(), ABSOLUTE_ERROR);
339 
340         final var std1 = estimator.getStandardDeviationAsMeasurement();
341         assertEquals(std, std1.getValue().doubleValue(), ABSOLUTE_ERROR);
342         assertEquals(AccelerationUnit.METERS_PER_SQUARED_SECOND, std1.getUnit());
343         final var std2 = new Acceleration(1.0, AccelerationUnit.FEET_PER_SQUARED_SECOND);
344         estimator.getStandardDeviationAsMeasurement(std2);
345         assertEquals(std1, std2);
346 
347         final var psd = v * timeInterval;
348         assertEquals(psd, estimator.getPsd(), ABSOLUTE_ERROR);
349 
350         final var rootPsd = Math.sqrt(psd);
351         assertEquals(rootPsd, estimator.getRootPsd(), ABSOLUTE_ERROR);
352         assertEquals(estimator.getRootPsd(), estimator.getAccelerometerBaseNoiseLevelRootPsd(), 0.0);
353 
354         assertEquals(windowSize, estimator.getNumberOfProcessedSamples());
355         assertEquals(windowSize, estimator.getNumberOfAddedSamples());
356         assertEquals(windowSize, estimator.getNumberOfSamplesInWindow());
357         assertFalse(estimator.isRunning());
358         assertTrue(estimator.isWindowFilled());
359 
360         assertEquals(1, start);
361         assertEquals(measurementAdded, windowSize);
362         assertEquals(1, windowFilled);
363         assertEquals(0, reset);
364 
365         // if we add more measurements, window filled is not called again
366         BodyKinematicsGenerator.generate(timeInterval, trueKinematics, errors, random, kinematics);
367         kinematics.getSpecificForceNormAsAcceleration(measurement);
368 
369         measurements.add(new Acceleration(measurement.getValue(), measurement.getUnit()));
370 
371         value = measurement.getValue().doubleValue();
372         estimator.addMeasurementAndProcess(value);
373 
374         assertEquals(windowSize + 1, estimator.getNumberOfProcessedSamples());
375         assertEquals(windowSize + 1, estimator.getNumberOfAddedSamples());
376         assertEquals(windowSize, estimator.getNumberOfSamplesInWindow());
377         assertTrue(estimator.isWindowFilled());
378 
379         assertEquals(measurements.size(), windowSize + 1);
380         assertEquals(measurements.get(1), estimator.getFirstWindowedMeasurement());
381         assertEquals(measurements.get(windowSize), estimator.getLastWindowedMeasurement());
382 
383         assertEquals(1, start);
384         assertEquals(windowSize + 1, measurementAdded);
385         assertEquals(1, windowFilled);
386         assertEquals(0, reset);
387 
388         // reset
389         assertTrue(estimator.reset());
390 
391         assertEquals(1, reset);
392 
393         assertNull(estimator.getFirstWindowedMeasurementValue());
394         assertNull(estimator.getFirstWindowedMeasurement());
395         assertFalse(estimator.getFirstWindowedMeasurement(null));
396         assertNull(estimator.getLastWindowedMeasurementValue());
397         assertNull(estimator.getLastWindowedMeasurement());
398         assertFalse(estimator.getLastWindowedMeasurement(null));
399         assertEquals(0.0, estimator.getAvg(), 0.0);
400         assertEquals(0.0, estimator.getVariance(), 0.0);
401         assertEquals(0.0, estimator.getStandardDeviation(), 0.0);
402         assertEquals(0.0, estimator.getPsd(), 0.0);
403         assertEquals(0.0, estimator.getRootPsd(), 0.0);
404         assertEquals(0.0, estimator.getAccelerometerBaseNoiseLevelRootPsd(), 0.0);
405         assertEquals(0, estimator.getNumberOfProcessedSamples());
406         assertEquals(0, estimator.getNumberOfAddedSamples());
407         assertEquals(0, estimator.getNumberOfSamplesInWindow());
408         assertFalse(estimator.isRunning());
409         assertFalse(estimator.isWindowFilled());
410     }

861     private void checkLocked(final WindowedAccelerationMeasurementNoiseEstimator estimator) {
862         assertTrue(estimator.isRunning());
863         assertThrows(LockedException.class, () -> estimator.setWindowSize(3));
864         assertThrows(LockedException.class, () -> estimator.setTimeInterval(0.0));
865         assertThrows(LockedException.class, () -> estimator.setTimeInterval(new Time(0.0, TimeUnit.SECOND)));
866         assertThrows(LockedException.class, () -> estimator.setListener(this));
867         assertThrows(LockedException.class, () -> estimator.addMeasurementAndProcess(0.0));
868         final var a = new Acceleration(0.0, AccelerationUnit.METERS_PER_SQUARED_SECOND);
869         assertThrows(LockedException.class, () -> estimator.addMeasurementAndProcess(a));
870         assertThrows(LockedException.class, () -> estimator.addMeasurement(0.0));
871         assertThrows(LockedException.class, () -> estimator.addMeasurement(a));
872         assertThrows(LockedException.class, estimator::reset);
873     }
