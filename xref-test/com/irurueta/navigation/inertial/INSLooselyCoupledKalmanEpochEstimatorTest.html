602     private static INSLooselyCoupledKalmanState estimate(
603             final double x, final double y, final double z, final double vx, final double vy, final double vz,
604             final INSLooselyCoupledKalmanState previousState, final double fx, final double fy, final double fz,
605             final double previousLatitude, final INSLooselyCoupledKalmanConfig config) throws AlgebraException {
606 
607         final var omegaIe = Utils.skewMatrix(new double[]{0.0, 0.0, Constants.EARTH_ROTATION_RATE});
608 
609         // SYSTEM PROPAGATION PHASE
610 
611         // 1. Determine transition matrix using (14.50) (first-order approx)
612         final var phiMatrix = Matrix.identity(15, 15);
613         phiMatrix.setSubmatrix(0, 0, 2, 2,
614                 phiMatrix.getSubmatrix(0, 0, 2, 2)
615                         .subtractAndReturnNew(omegaIe.multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS)));
616 
617         final var estCbeOld = previousState.getBodyToEcefCoordinateTransformationMatrix();
618         phiMatrix.setSubmatrix(0, 12, 2, 14,
619                 estCbeOld.multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS));
620 
621         final var measFibb = Matrix.newFromArray(new double[]{fx, fy, fz});
622         phiMatrix.setSubmatrix(3, 0, 5, 2,
623                 Utils.skewMatrix(estCbeOld.multiplyAndReturnNew(measFibb))
624                         .multiplyByScalarAndReturnNew(-TIME_INTERVAL_SECONDS));
625 
626         phiMatrix.setSubmatrix(3, 3, 5, 5,
627                 phiMatrix.getSubmatrix(3, 3, 5, 5)
628                         .subtractAndReturnNew(omegaIe.multiplyByScalarAndReturnNew(2.0 * TIME_INTERVAL_SECONDS)));
629 
630         final var sinPrevLat = Math.sin(previousLatitude);
631         final var cosPrevLat = Math.cos(previousLatitude);
632         final var sinPrevLat2 = sinPrevLat * sinPrevLat;
633         final var cosPrevLat2 = cosPrevLat * cosPrevLat;
634 
635         final var geocentricRadius = Constants.EARTH_EQUATORIAL_RADIUS_WGS84
636                 / Math.sqrt(1.0 - Math.pow(Constants.EARTH_ECCENTRICITY * sinPrevLat, 2.0))
637                 * Math.sqrt(cosPrevLat2
638                 + Math.pow(1.0 - Constants.EARTH_ECCENTRICITY * Constants.EARTH_ECCENTRICITY, 2.0)
639                 * sinPrevLat2);
640 
641         final var prevX = previousState.getX();
642         final var prevY = previousState.getY();
643         final var prevZ = previousState.getZ();
644         final var gravity = ECEFGravityEstimator.estimateGravityAndReturnNew(prevX, prevY, prevZ);
645         final var g = gravity.asMatrix();
646 
647         final var estRebeOld = Matrix.newFromArray(new double[]{prevX, prevY, prevZ});
648 
649         final var gScaled = g.multiplyByScalarAndReturnNew(-2.0 * TIME_INTERVAL_SECONDS / geocentricRadius);
650         final var estRebeOldTrans = estRebeOld.transposeAndReturnNew();
651         final var previousPositionNorm = Math.sqrt(prevX * prevX + prevY * prevY + prevZ * prevZ);
652         final var estRebeOldTransScaled = estRebeOldTrans.multiplyByScalarAndReturnNew(1.0 / previousPositionNorm);
653         phiMatrix.setSubmatrix(3, 6, 5, 8,
654                 gScaled.multiplyAndReturnNew(estRebeOldTransScaled));
655 
656         phiMatrix.setSubmatrix(3, 9, 5, 11,
657                 estCbeOld.multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS));
658 
659         phiMatrix.setSubmatrix(6, 3, 8, 5,
660                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(TIME_INTERVAL_SECONDS));
661 
662         // 2. Determine approximate system noise covariance matrix using (14.82)
663         final var qPrimeMatrix = new Matrix(15, 15);
664         qPrimeMatrix.setSubmatrix(0, 0, 2, 2,
665                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
666                         config.getGyroNoisePSD() * TIME_INTERVAL_SECONDS));
667         qPrimeMatrix.setSubmatrix(3, 3, 5, 5,
668                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
669                         config.getAccelerometerNoisePSD() * TIME_INTERVAL_SECONDS));
670         qPrimeMatrix.setSubmatrix(9, 9, 11, 11,
671                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
672                         config.getAccelerometerBiasPSD() * TIME_INTERVAL_SECONDS));
673         qPrimeMatrix.setSubmatrix(12, 12, 14, 14,
674                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
675                         config.getGyroBiasPSD() * TIME_INTERVAL_SECONDS));
676 
677         // 3. Propagate state estimates using (3.14) noting that all states are zero
678         final var xEstPropagated = new Matrix(15, 1);
679 
680         // 4. Propagate state estimation error covariance matrix using (3.46)
681         final var pMatrixOld = previousState.getCovariance();
682         final var phiTrans = phiMatrix.transposeAndReturnNew();
683         final var halfQ = qPrimeMatrix.multiplyByScalarAndReturnNew(0.5);
684         final var tmp1 = pMatrixOld.addAndReturnNew(halfQ);
685         final var pMatrixPropagated = phiMatrix.multiplyAndReturnNew(tmp1);
686         pMatrixPropagated.multiply(phiTrans);
687         pMatrixPropagated.add(halfQ);
688 
689         // MEASUREMENT UPDATE PHASE
690 
691         // 5. Set up a measurement matrix using (14.115)
692         final var hMatrix = new Matrix(6, 15);
693         hMatrix.setSubmatrix(0, 6, 2, 8,
694                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(-1.0));
695         hMatrix.setSubmatrix(3, 3, 5, 5,
696                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(-1.0));
697 
698         // 6. Set up a measurement noise covariance matrix assuming all components of
699         // GNSS position and velocity are independent and have equal variance
700         final var rMatrix = new Matrix(6, 6);
701         rMatrix.setSubmatrix(0, 0, 2, 2,
702                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
703                         Math.pow(config.getPositionNoiseSD(), 2.0)));
704         rMatrix.setSubmatrix(0, 3, 2, 5,
705                 new Matrix(3, 3));
706         rMatrix.setSubmatrix(3, 0, 5, 2,
707                 new Matrix(3, 3));
708         rMatrix.setSubmatrix(3, 3, 5, 5,
709                 Matrix.identity(3, 3).multiplyByScalarAndReturnNew(
710                         Math.pow(config.getVelocityNoiseSD(), 2.0)));
711 
712         // 7. Calculate Kalman gain using (3.21)
713         final var tmp2 = hMatrix.multiplyAndReturnNew(pMatrixPropagated).multiplyAndReturnNew(
714                 hMatrix.transposeAndReturnNew());
715         tmp2.add(rMatrix);
716         final var kMatrix = pMatrixPropagated.multiplyAndReturnNew(hMatrix.transposeAndReturnNew())
717                 .multiplyAndReturnNew(Utils.inverse(tmp2));
718 
719         // 8. Formulate measurement innovations using (14.102), noting that zero-lever arm is assumed here
720         final var deltaZ = new Matrix(6, 1);
721 
722         final var prevVx = previousState.getVx();
723         final var prevVy = previousState.getVy();
724         final var prevVz = previousState.getVz();
725         final var estVebeOld = Matrix.newFromArray(new double[]{prevVx, prevVy, prevVz});
726 
727         final var gnssRebe = Matrix.newFromArray(new double[]{x, y, z});
728         final var gnssVebe = Matrix.newFromArray(new double[]{vx, vy, vz});
729 
730         deltaZ.setSubmatrix(0, 0, 2, 0,
731                 gnssRebe.subtractAndReturnNew(estRebeOld));
732         deltaZ.setSubmatrix(3, 0, 5, 0,
733                 gnssVebe.subtractAndReturnNew(estVebeOld));
734 
735         // 9. Update state estimates using (3.24)
736         final var xEstNew = xEstPropagated.addAndReturnNew(kMatrix.multiplyAndReturnNew(deltaZ));
737 
738         // 10. Update state estimation error covariance matrix using (3.25)
739         final var pMatrixNew = (Matrix.identity(15, 15)
740                 .subtractAndReturnNew(kMatrix.multiplyAndReturnNew(hMatrix)))
741                 .multiplyAndReturnNew(pMatrixPropagated);
742 
743         // CLOSED-LOOP CORRECTION
744 
745         // Correct attitude, velocity, and position using (14.7-9)
746         final var estCbeNew = (Matrix.identity(3, 3).subtractAndReturnNew(
747                 Utils.skewMatrix(
748                         xEstNew.getSubmatrix(0, 0, 2, 0))))
749                 .multiplyAndReturnNew(estCbeOld);
750         final var estVebeNew = estVebeOld.subtractAndReturnNew(
751                 xEstNew.getSubmatrix(3, 0, 5, 0));
752         final var estRebeNew = estRebeOld.subtractAndReturnNew(
753                 xEstNew.getSubmatrix(6, 0, 8, 0));
754 
755         // Update IMU bias estimates
756         final var estIMUbiasOld = Matrix.newFromArray(new double[]{
757                 previousState.getAccelerationBiasX(),
758                 previousState.getAccelerationBiasY(),
759                 previousState.getAccelerationBiasZ(),
760                 previousState.getGyroBiasX(),
761                 previousState.getGyroBiasY(),
762                 previousState.getGyroBiasZ()
763         });
764         final var estIMUbiasNew = estIMUbiasOld.addAndReturnNew(
765                 xEstNew.getSubmatrix(9, 0, 14, 0));
766 
767         return new INSLooselyCoupledKalmanState(estCbeNew,
768                 estVebeNew.getElementAtIndex(0),
769                 estVebeNew.getElementAtIndex(1),
770                 estVebeNew.getElementAtIndex(2),
771                 estRebeNew.getElementAtIndex(0),
772                 estRebeNew.getElementAtIndex(1),
773                 estRebeNew.getElementAtIndex(2),
774                 estIMUbiasNew.getElementAtIndex(0),
775                 estIMUbiasNew.getElementAtIndex(1),
776                 estIMUbiasNew.getElementAtIndex(2),
777                 estIMUbiasNew.getElementAtIndex(3),
778                 estIMUbiasNew.getElementAtIndex(4),
779                 estIMUbiasNew.getElementAtIndex(5),
780                 pMatrixNew);
781     }
