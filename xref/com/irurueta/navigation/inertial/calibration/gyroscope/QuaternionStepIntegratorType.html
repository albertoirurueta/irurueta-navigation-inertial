16  package com.irurueta.navigation.inertial.calibration.gyroscope;
17  
18  /**
19   * Indicates type of quaternion integrator step.
20   * Different types exist with different levels of accuracy and computation complexity.
21   */
22  public enum QuaternionStepIntegratorType {
23      /**
24       * Performs quaternion integration using Euler's method, which is the less accurate and has the
25       * smallest computational complexity.
26       */
27      EULER_METHOD,
28  
29      /**
30       * Performs quaternion integration using mid-point algorithm, which offers a medium accuracy and
31       * computational complexity.
32       */
33      MID_POINT,
34  
35      /**
36       * Performs quaternion integration using Runge-Kutta of 4th order (aka RK4) algorithm, which
37       * offers high accuracy at the expense of higher computational complexity.
38       */
39      RUNGE_KUTTA,
40  
41      /**
42       * Performs quaternion integration Based on Young Soo Suh. "Orientation estimation using a quaternion-based
43       * indirect Kalman filter with adaptive estimation of external acceleration". 2010.
44       * This method can achieve higher accuracy than Runge-Kutta
45       */
46      SUH,
47  
48      /**
49       * Performs quaternion integration based on Trawny, N. "Indirect Kalman Filter for 3D Attitude Estimation". 2005,
50       * which offers a medium accuracy and computational complexity.
51       */
52      TRAWNY,
53  
54      /**
55       * Performs quaternion integration based on Yuan, S. "Quaternion-based Unscented Kalman Filter for Real-time". 2015,
56       * which offers a medium accuracy and computational complexity.
57       */
58      YUAN
59  }
