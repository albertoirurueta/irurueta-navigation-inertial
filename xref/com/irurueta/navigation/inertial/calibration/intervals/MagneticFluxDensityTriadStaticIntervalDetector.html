16  package com.irurueta.navigation.inertial.calibration.intervals;
17  
18  import com.irurueta.navigation.inertial.calibration.MagneticFluxDensityTriad;
19  import com.irurueta.navigation.inertial.calibration.noise.AccumulatedMagneticFluxDensityTriadNoiseEstimator;
20  import com.irurueta.navigation.inertial.calibration.noise.WindowedMagneticFluxDensityTriadNoiseEstimator;
21  import com.irurueta.units.MagneticFluxDensity;
22  import com.irurueta.units.MagneticFluxDensityConverter;
23  import com.irurueta.units.MagneticFluxDensityUnit;
24  
25  /**
26   * This detector is in charge of determining when a static period of
27   * magnetic flux density measurements starts and finishes.
28   * Static periods are periods of time where the device is considered
29   * to remain static (no movement applied to it).
30   */
31  public class MagneticFluxDensityTriadStaticIntervalDetector extends
32          TriadStaticIntervalDetector<MagneticFluxDensityUnit, MagneticFluxDensity, MagneticFluxDensityTriad,
33                  MagneticFluxDensityTriadStaticIntervalDetector,
34                  MagneticFluxDensityTriadStaticIntervalDetectorListener> {
35  
36      /**
37       * Constructor.
38       */
39      public MagneticFluxDensityTriadStaticIntervalDetector() {
40          super(new WindowedMagneticFluxDensityTriadNoiseEstimator(),
41                  new AccumulatedMagneticFluxDensityTriadNoiseEstimator());
42      }
43  
44      /**
45       * Constructor.
46       *
47       * @param listener listener to handle events generated by this detector.
48       */
49      public MagneticFluxDensityTriadStaticIntervalDetector(
50              final MagneticFluxDensityTriadStaticIntervalDetectorListener listener) {
51          super(new WindowedMagneticFluxDensityTriadNoiseEstimator(),
52                  new AccumulatedMagneticFluxDensityTriadNoiseEstimator(), listener);
53      }
54  
55      /**
56       * Converts provided magnetic flux density value expressed in provided unit
57       * to Teslas (T).
58       *
59       * @param value value to be converted.
60       * @param unit  unit of value to be converted.
61       * @return converted value.
62       */
63      @Override
64      protected double convertMeasurement(final double value, final MagneticFluxDensityUnit unit) {
65          return MagneticFluxDensityConverter.convert(value, unit, MagneticFluxDensityUnit.TESLA);
66      }
67  
68      /**
69       * Creates a magnetic flux density instance using provided value and unit.
70       *
71       * @param value value of measurement.
72       * @param unit  unit of value.
73       * @return created magnetic flux density.
74       */
75      @Override
76      protected MagneticFluxDensity createMeasurement(final double value, final MagneticFluxDensityUnit unit) {
77          return new MagneticFluxDensity(value, unit);
78      }
79  
80      /**
81       * Gets default unit for magnetic flux density, which is Teslas (T).
82       *
83       * @return default unit for magnetic flux density.
84       */
85      @Override
86      protected MagneticFluxDensityUnit getDefaultUnit() {
87          return MagneticFluxDensityUnit.TESLA;
88      }
89  
90      /**
91       * Creates a magnetic flux density triad.
92       *
93       * @param valueX x-coordinate value.
94       * @param valueY y-coordinate value.
95       * @param valueZ z-coordinate value.
96       * @param unit   unit of values.
97       * @return created triad.
98       */
99      @Override
100     protected MagneticFluxDensityTriad createTriad(
101             final double valueX, final double valueY, final double valueZ, final MagneticFluxDensityUnit unit) {
102         return new MagneticFluxDensityTriad(unit, valueX, valueY, valueZ);
103     }
104 }
